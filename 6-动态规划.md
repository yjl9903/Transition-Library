# 动态规划

## SOS DP

求 $F[mask]=\sum_{s \subseteq mask} a[s]$。

令 $dp(mask,i)$ 表示 $mask$ 的前 $i$ 位不同的子集权值和，转移为：
$$
dp(mask,i)=
\begin{cases}
dp(mask,i-1) & i^{th} \text{bit off} \\
dp(mask,i-1)+dp(mask \oplus 2^i,i-1) & i^{th} \text{bit on}
\end{cases}
$$

```c++
//iterative version
for (int mask = 0; mask < (1 << N); mask++) {
    dp[mask][-1] = A[mask]; //handle base case separately (leaf states)
    for (int i = 0; i < N; i++) {
        if (mask & (1 << i))
            dp[mask][i] = dp[mask][i - 1] + dp[mask ^ (1 << i)][i - 1];
        else
            dp[mask][i] = dp[mask][i - 1];
    }
    F[mask] = dp[mask][N - 1];
}

//memory optimized, super easy to code.
for (int i = 0; i < (1 << N); i++)
    F[i] = A[i];
for (int i = 0; i < N; i++)
  for (int mask = 0; mask < (1 << N); mask++) {
    if (mask & (1 << i))
      F[mask] += F[mask ^ (1 << i)];
  }
```

## 插头 DP

### 二进制多回路

```C++
// 有障碍多条回路方案数
#include<bits/stdc++.h>
using namespace std;
#define ll long long
const int N=12;
int t,m,n,s[N][N];
ll ans,dp[N][N][1<<N],bin[N+1];

void init(){
    bin[0]=1;
    for(int i=1; i<=N; i++)
        bin[i]=bin[i-1]<<1;
    return;
}

int main(){
    scanf("%d",&t);
    init();
    for(int tt=1; tt<=t; tt++){
        scanf("%d%d",&n,&m);
        for(int i=1; i<=n; i++){
            for(int j=1; j<=m; j++){
                scanf("%d",&s[i][j]);
            }
        }
        memset(dp,0,sizeof(dp));
        dp[1][0][0]=1;
        for(int i=1; i<=n; i++){
            for(int j=1; j<=m; j++){
                for(int k=0; k<bin[m+1]; k++){
                    int pa=bin[j-1],pb=bin[j];
                    if(s[i][j]==1){
                        dp[i][j][k]+=dp[i][j-1][k^pa^pb];
                        if(((k&pa)==0)^((k&pb)==0))
                            dp[i][j][k]+=dp[i][j-1][k];
                    }else{
                        if(!(k&pa) && !(k&pb))
                            dp[i][j][k]=dp[i][j-1][k];
                        else
                            dp[i][j][k]=0;
                    }
                }
            }
            if(i!=n){
                for(int k=0; k<bin[m]; k++){
                    dp[i+1][0][k<<1]=dp[i][m][k];
                }
            }
        }
        printf("Case %d: There are %lld ways to eat the trees.\n",tt,dp[n][m][0]);
    }
    return 0;
}

```

### 三进制单回路

```C++
// 无障碍哈密尔顿回路方案数
#include<bits/stdc++.h>
using namespace std;
#define ll unsigned long long

const int N=10,M=21;
const ll PO=1e18;

struct number{
    ll x,y;

    number operator + (number a){
        return (number){x+a.x+(y+a.y)/PO,(y+a.y)%PO};
    }

    number& operator += (number a){
        *this=*this+a;
        return *this;
    }
    void print(){
        if (x) printf("%llu%018llu\n", x, y);
        else printf("%llu\n", y);
    }
};


int t,n,m;
int code[1<<(N*2+2)],decode[177150],tot=0;
number dp[N+1][177150],tmp[177150];
ll bin[N+2],pow3[N+2];


void addcode(int x){
    int tmpx=x;
    while(tmpx){
        if((tmpx&3)==3) return;
        tmpx>>=2;
    }
    code[x]=tot;
    decode[tot]=x;
    tot++;
    return;
}

void init(){
    pow3[0]=1;
    bin[0]=1;
    for(int i=1; i<=N+1; i++){
        bin[i]=bin[i-1]<<2;
        pow3[i]=pow3[i-1]*3;
    }
    for(int i=0; i<(1<<(N*2+2)); i++){
        addcode(i);
    }
    //cout<<tot<<endl;
    return;
}

int main(){
    freopen("postman.in","r",stdin);
    freopen("postman.out","w",stdout);
    init();
    scanf("%d%d",&n,&m);
    dp[0][0]=(number){0,1};
    for(int i=1; i<=m; i++){
        for(int j=1; j<=n; j++){
            //cout<<i<<" "<<j<<endl;
            for(int myk=0; myk<pow3[n+1]; myk++){
                int k=decode[myk];
                //cout<<myk<<" "<<k<<endl;
                int pa=bin[j-1],pb=bin[j];
                int ba=(k>>(j*2-2))%4,bb=(k>>(j*2))%4;
                //cout<<pa<<" "<<pb<<" "<<ba<<" "<<bb<<endl;
                if(!ba && !bb){
                    dp[j][code[k|pa|(pb<<1)]]+=dp[j-1][code[k]];
                }
                else if(!ba || !bb){
                    dp[j][code[k]]+=dp[j-1][code[k]];
                    dp[j][code[k^(pa*ba+pb*bb)^(pa*bb+pb*ba)]]+=dp[j-1][code[k]];
                }
                else if(ba==1 && bb==1){
                    int numl=1;
                    for(int h=j+1; h<=n; h++){
                        if(k&bin[h]) numl++;
                        if(k&(bin[h]<<1)) numl--;
                        if(!numl){
                            dp[j][code[k^(bin[h]*3)^pa^pb]]+=dp[j-1][code[k]];
                            break;
                        }
                    }
                }
                else if(ba==2 && bb==2){
                    int numr=1;
                    for(int h=j-2; h>=0; h--){
                        if(k&bin[h]) numr--;
                        if(k&(bin[h]<<1)) numr++;
                        if(!numr){
                            dp[j][code[k^(bin[h]*3)^(pa<<1)^(pb<<1)]]+=dp[j-1][code[k]];
                            break;
                        }
                    }
                }
                else if(ba==1 && bb==2){
                    if(i==m && j==n)
                        dp[j][code[k^pa^(pb<<1)]]+=dp[j-1][code[k]];
                }
                else if(ba==2 && bb==1){
                    dp[j][code[k^(pa<<1)^pb]]+=dp[j-1][code[k]];
                }
            }
        }
        if(i!=m){
            for(int myk=0; myk<pow3[n]; myk++){
                int k=decode[myk];
                tmp[code[k]]=dp[n][code[k]];
            }
            memset(dp,0,sizeof(dp));
            for(int myk=0; myk<pow3[n]; myk++){
                int k=decode[myk];
                dp[0][code[k<<2]]=tmp[code[k]];
            }
        }
        //cout<<i<<endl;
    }
    number ans=dp[n][code[0]]+dp[n][code[0]];
    if(ans.x==0 && ans.y==0) ans.y=1;
    ans.print();
    return 0;
}
```
